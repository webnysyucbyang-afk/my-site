1. Let a be an array of integers. Write a recursive C function
   to compute the sum of the elements of the array.

   {
        if(size == 0)
          return 0;
        else
          return a[size-1] + sum(a, size - 1);

2. Write a recursive C function to determine the maximum
   of the contents of all the nodes in a binary tree.
Solution:
       sturct nodetype {
           int info;
           struct nodetype *left;
           struct nodetype *right;
       }
       int max(struct nodetype *p)
       {
          int a,b;

          if(p==NULL)
             return(-MAXINT); //MAXINT is infinite
          else {
            a=max(p->left);
            b=max(p->right);
            if (a>=b)
               return (p->info >= a ? p->info : a)
            else
               return (p->info >= b ? p->info : b)
           }
       }   /* end of max() */
